let cenaAtual = "inicio";

function atualizarCena() {
    const cena = cenas[cenaAtual];
    
    document.getElementById("cena-texto").innerText = cena.texto;
    document.getElementById("cena-imagem").src = cena.imagem;
    
    const areaOpcoes = document.getElementById("opcoes");
    areaOpcoes.innerHTML = ""; // Limpa botões antigos
    
    cena.opcoes.forEach(op => {
        const botao = document.createElement("button");
        botao.innerText = op.texto;
        if (op.classe) botao.classList.add(op.classe);
        botao.addEventListener("click", () => {
            cenaAtual = op.proxima;
            atualizarCena();
        });
        areaOpcoes.appendChild(botao);
    });
}

// Inicia o jogo
atualizarCena();
